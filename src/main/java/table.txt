CREATE TABLE USER( ##유저 테이블
   USER_NUM INT PRIMARY KEY AUTO_INCREMENT, ##유저번호
   USER_ID VARCHAR(20), ##아이디
   USER_PW VARCHAR(20), ##비밀번호
   USER_NAME VARCHAR(20), ##이름
   USER_AGE VARCHAR(20) NOT NULL, ##나이
   USER_GEN VARCHAR(1), ##성별
   USER_EMAIL VARCHAR(50), ##이메일
   USER_PHONE VARCHAR(30), ##전화번호
   USER_SALARY INT NOT NULL, ##연봉
   USER_SIZE INT NOT NULL, ##집 크기
   USER_STAY INT NOT NULL, ##집에있는시간
   USER_LIKE VARCHAR(50) NOT NULL, ##집에있는시간
   USER_ING VARCHAR(1) NOT NULL, ##집에있는시간
   REC_LIST_NUM INT 
); ##추천리스트 번호



CREATE TABLE FREE_LIST( ##자유게시판
   FREE_LIST_NUM INT PRIMARY KEY AUTO_INCREMENT, ##유저 리스트번호
   FREE_LIST_TITLE VARCHAR(50), ##제목
   FREE_LIST_CONTENT VARCHAR(2000), ##내용
   FREE_LIST_TYPE VARCHAR(20), ##종류
   FREE_LIST_DATE TIMESTAMP DEFAULT NOW(), ##날짜
   FREE_LIST_VIEWS INT DEFAULT 0, ##조회수
   FREE_LIST_WRITER VARCHAR(20), ##아이디
   USER_NUM INT
);  ##유저번호



CREATE TABLE ADOPTLIST(##입양신청이력
   ADOPT_LIST_NUM INT PRIMARY KEY AUTO_INCREMENT, ##입양이력번호
   USER_NUM INT,
   ADOPT_LIST_NAME VARCHAR(30), ##입양동물이름
   ADOPT_LIST_TYPE VARCHAR(20), ##입양동물종류
   ADOPT_LIST_DATE TIMESTAMP DEFAULT NOW(), ##입양날짜
   ADOPT_LIST_PROGRESS VARCHAR(20) 
); ##입양진행상황



CREATE TABLE ANIMAL( ##동물 테이블
ANIMAL_NUM INT PRIMARY KEY AUTO_INCREMENT, ##동물 번호
ANIMAL_TYPE VARCHAR(20), ##동물종류
ANIMAL_SIZE VARCHAR(1), ##동물크기
ANIMAL_NAME VARCHAR(30), ##동물이름
ANIMAL_GENDER VARCHAR(1), ##동물성별
ANIMAL_AGE INT NOT NULL, ##동물나이
ANIMAL_CONTENT VARCHAR(500), ##동물참조내용
ANIMAL_DISEASE VARCHAR(1), ##동물질병유무
ANIMAL_NEUTRAL VARCHAR(20), ##중성화여부
ANIMAL_COST INT NOT NULL, ##유지비
ANIMAL_TIME INT NOT NULL, ##돌봄필요시간 
ANIMAL_PIC VARCHAR(1000) ,##동물사진
ANIMAL_DATE TIMESTAMP DEFAULT NOW(), ##등록일
ADMIN_NUM INT, ##관리자번호
LIST_NUM INT ); ##동물리스트번호



CREATE TABLE COMMENT ( ##댓글
   COMMENT_NUM INT PRIMARY KEY AUTO_INCREMENT, ##댓글번호
   USER_NUM INT, 
   ANIMAL_NUM INT,
   USER_LIST_NUM INT,
   COMMENT_TYPE VARCHAR(20), ##게시판종류
   COMMENT_CONTENT VARCHAR(2000), ##댓글내용
   COMMENT_DATE TIMESTAMP DEFAULT NOW() 
); ##날짜



##관리자 게시판
CREATE TABLE ADMIN_LIST(
   AD_LIST_NUM INT PRIMARY KEY AUTO_INCREMENT,
   AD_LIST_TILTE VARCHAR(300),
   AD_LIST_WRITER VARCHAR(50),
   AD_LIST_CONTENT VARCHAR(2000),
   AD_LIST_DATE TIMESTAMP DEFAULT NOW(), 
   ADMIN_ID VARCHAR(50),
   ADMIN_PW VARCHAR(20)
);
